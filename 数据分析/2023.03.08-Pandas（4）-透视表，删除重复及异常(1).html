#pandas不能直接处理异常值
#所以，将异常值置为空值，再进行处理

#定义替换函数，形参x是系列
def replace(x):
    QU=x.quantile(0.75)
    QL=x.quantile(0.25)
    IQR=QU-QL
    index=(x>(QU+1.5*IQR)) |(x< (QL-1.5*IQR))
    x[index]=np.nan
    return x

orders['amounts'].isnull().sum()  #0个缺失值

orders['amounts']=replace(orders['amounts'])

orders['amounts'].isnull().sum()  #173个缺失值

#定义替换函数，形参 x 是系列
def replace1(x):
    sigma=x.std()
    u=x.mean()
    LOW = u-3*sigma
    HIGH = u+3*sigma
    index=(x > HIGH) |(x < LOW)
    x[index]=np.nan
    return x
